val hello = sc.parallelize(Seq("Hello World"))

#Arithmetic
val df = spark.range(1, 11).toDF("number")
val results = df.withColumn("plus_ten", $"number" + 10)
                .withColumn("squared", $"number" * $"number")
                .withColumn("half", $"number" / 2.0)
                .withColumn("is_even", $"number" % 2 === 0)
results.show()

#List Array
val numbers = List(1,2,3,4,5)
val rdd = spark.sparkContext.parallelize(numbers)
rdd.collect()

#Word Count
val text = List(&quot;hello spark&quot;, &quot;hello scala&quot;)
val rdd = spark.sparkContext.parallelize(text)
val words = rdd.flatMap(line =&gt; line.split(&quot; &quot;))
val pairs = words.map(word =&gt; (word,1))
val counts = pairs.reduceByKey((a,b) =&gt; a+b)
counts.collect()
